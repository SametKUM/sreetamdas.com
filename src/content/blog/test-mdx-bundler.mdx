```ts highlight="1,3-4"
const htmlEscapes = {
  "&": "&amp;",
  "<": "&lt;",
  ">": "&gt;",
  // eslint-disable-next-line quotes
  '"': "&quot;",
  "'": "&#39;",
} as const;

function escapeHtml(html: string) {
  return html.replace(
    /[&<>"']/g,
    (chr) => htmlEscapes[chr as keyof typeof htmlEscapes]
  );
}
```
